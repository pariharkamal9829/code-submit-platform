// Firebase Firestore Security Rules
// Copy these rules to your Firebase Console -> Firestore Database -> Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow read/write access to all authenticated users for now
    // You can make this more restrictive later
    
    // Submissions collection - authenticated users can read/write their own
    match /submissions/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Admins collection - authenticated users can read, only admins can write
    match /admins/{document} {
      allow read: if request.auth != null;
      // Allow write for initial admin creation and existing admins
      allow write: if request.auth != null && (
        // Allow creating the first admin
        !exists(/databases/$(database)/documents/admins/$(document)) ||
        // Or if user is already an admin
        exists(/databases/$(database)/documents/admins/$(request.auth.token.email.replace('[^a-zA-Z0-9]', '_')))
      );
    }
    
    // For now, allow all authenticated users (TEMPORARY - for debugging)
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
